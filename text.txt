SECURE_CLOUD_ENDPOINT = "https://kvdb.io/K99vVnZgYqN7Cg8n5vP9a7/nexus_request"
NOTES_JSON_FILE = "nexus_notes.json"

class NexusLearnUnifiedApp:

    def ensure_json_notes_exists(self):
        if not os.path.exists(NOTES_JSON_FILE):
            with open(NOTES_JSON_FILE, "w") as f:
                json.dump({}, f)

    def start_cloud_polling_engine(self):
        while True:
            time.sleep(2.0)
            try:
                response = requests.get(SECURE_CLOUD_ENDPOINT, timeout=3)
                if response.status_code == 200 and response.text.strip():
                    request_data = response.json()
                    req_time = request_data.get("timestamp", 0)
                    
                    if req_time > self.last_processed_time:
                        self.last_processed_time = req_time
                        req_type = request_data.get("type")
                        
                        if req_type == "TEACHER_LOGIN":
                            self.root.after(0, self.handle_web_teacher_auth, request_data)
                        elif req_type == "STUDENT_LOGIN":
                            self.root.after(0, self.handle_web_student_auth, request_data)
                        elif req_type == "STUDENT_ACTION" and request_data.get("code") == self.active_session_pin:
                            self.root.after(0, self.process_web_student_action, request_data.get("name"), request_data.get("action"))
                        elif req_type == "CBT_SUBMISSION" and request_data.get("code") == self.active_session_pin:
                            self.root.after(0, self.process_web_cbt_submission, request_data.get("name"), request_data.get("answer"))
            except Exception:
                pass

    def handle_web_teacher_auth(self, data):
        self.student_name = data.get("username")
        if hasattr(self, 'load_frame') and self.load_frame.winfo_exists(): self.load_frame.destroy()
        if hasattr(self, 'gateway_frame') and self.gateway_frame.winfo_exists(): self.gateway_frame.destroy()
        self.run_teacher_10s_transition()
        self.confirm_handshake_to_cloud(self.student_name)

    def handle_web_student_auth(self, data):
        target_code = data.get("code")
        if hasattr(self, 't_roster') and len(self.t_roster.get_children()) >= self.max_students_limit:
            return

        if hasattr(self, 't_roster') and self.t_roster.winfo_exists() and target_code == self.active_session_pin:
            self.root.after(0, self.register_live_student_node, data.get("name"))
        elif not hasattr(self, 't_roster'):
            self.student_name = data.get("name")
            self.active_session_pin = target_code
            if hasattr(self, 'gateway_frame') and self.gateway_frame.winfo_exists(): self.gateway_frame.destroy()
            self.build_student_workspace()
        self.confirm_handshake_to_cloud(data.get("name"))

    def process_web_student_action(self, name, action):
        if hasattr(self, 't_roster'):
            action_labels = {
                "EMOTE_RAISE": "🖐️ Hand Raised (Awaiting Prompt)",
                "EMOTE_CLAP": "👏 Sent Performance Applaud"
            }
            mapped_text = action_labels.get(action, "Active Workspace Feed Update")
            self.register_live_student_node(name, status=mapped_text)

    def process_web_cbt_submission(self, name, answer):
        if hasattr(self, 't_roster'):
            self.register_live_student_node(name, status="CBT Score Transmitted Securely", grade=f"Locked: {answer}")

    def confirm_handshake_to_cloud(self, username):
        def run():
            try:
                confirm_url = f"https://kvdb.io/K99vVnZgYqN7Cg8n5vP9a7/nexus_confirm_{username}"
                requests.post(confirm_url, json={"status": "ACTIVE_IN_APP"}, timeout=3)
            except Exception:
                pass
        threading.Thread(target=run, daemon=True).start()

    def push_global_broadcast(self, payload):
        def task():
            try:
                payload["timestamp"] = int(time.time() * 1000)
                requests.post(f"https://kvdb.io/K99vVnZgYqN7Cg8n5vP9a7/nexus_broadcast_{self.active_session_pin}", json=payload, timeout=3)
            except Exception:
                pass
        threading.Thread(target=task, daemon=True).start()

    def register_live_student_node(self, name, status="Node Hook Established", grade="No Evaluation Records"):
        if hasattr(self, 't_roster'):
            for item in self.t_roster.get_children():
                if self.t_roster.item(item)['values'][0] == name:
                    self.t_roster.item(item, values=(name, status, grade))
                    return
            self.t_roster.insert("", tk.END, values=(name, status, grade))

    def init_shared_database(self):
        self.conn = sqlite3.connect("nexuslearn_system.db", check_same_thread=False)
        self.cursor = self.conn.cursor()
        self.cursor.execute("""
            CREATE TABLE IF NOT EXISTS teachers (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                username TEXT UNIQUE NOT NULL,
                password TEXT NOT NULL,
                auth_pin TEXT UNIQUE NOT NULL,
                payment_status TEXT DEFAULT 'Unpaid',
                expiration_timestamp DATETIME
            )
        """)
        self.cursor.execute("""
            CREATE TABLE IF NOT EXISTS chat_logs (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                session_id TEXT,
                sender TEXT,
                recipient TEXT,
                message TEXT,
                timestamp DATETIME DEFAULT CURRENT_TIMESTAMP
            )
        """)
        self.conn.commit()

    def hub_generate_pin(self):
        digits = "".join(random.choices(string.digits, k=4))
        letters = "".join(random.choices(string.ascii_uppercase, k=2))
        self.active_session_pin = digits + letters
        self.lbl_hub_pin.config(text=f"INVITATION PIN: {self.active_session_pin}")

    def hub_load_audio_file(self):
        path = filedialog.askopenfilename(filetypes=[("Class Audio Track", "*.mp3 *.wav *.ogg")])
        if path:
            self.current_playing_track = os.path.basename(path)
            self.lbl_music_track.config(text=f"Loaded: {self.current_playing_track}", fg="#5BC0BE")

    def hub_play_audio_mock(self):
        if self.current_playing_track:
            self.lbl_music_track.config(text=f"🔊 Playing: {self.current_playing_track}", fg="#FFB703")

    def hub_stop_audio_mock(self):
        if self.current_playing_track:
            self.lbl_music_track.config(text=f"Loaded: {self.current_playing_track}", fg="#5BC0BE")

    def hub_save_note_to_json(self):
        title = self.ent_note_title.get().strip() or "Untitled"
        content = self.hub_notepad.get("1.0", tk.END).strip()
        try:
            with open(NOTES_JSON_FILE, "r") as f:
                data = json.load(f)
            data[title] = content
            with open(NOTES_JSON_FILE, "w") as f:
                json.dump(data, f, indent=4)
        except Exception:
            pass

    def hub_load_note_from_json(self):
        title = self.ent_note_title.get().strip()
        try:
            with open(NOTES_JSON_FILE, "r") as f:
                data = json.load(f)
            if title in data:
                self.hub_notepad.delete("1.0", tk.END)
                self.hub_notepad.insert(tk.END, data[title])
        except Exception:
            pass

    def hub_extract_image_text(self):
        path = filedialog.askopenfilename(filetypes=[("Image Components", "*.png *.jpg *.jpeg")])
        if path:
            filename = os.path.basename(path)
            extracted_words = f"\n\n[Extracted Vectors from Document Frame: {filename}]\nTopic: Advanced Technical Lecture Segment\nContent Module Sample: Unified Sync Node Active Parameters."
            self.hub_notepad.insert(tk.END, extracted_words)

    def teacher_launch_class(self):
        self.current_class_name = self.ent_cname.get().strip()
        if not self.current_class_name or not self.active_session_pin:
            return
        
        self.cursor.execute("INSERT INTO chat_logs (session_id, sender, recipient, message) VALUES (?, 'SYSTEM', 'BROADCAST', ?)",
                            (self.active_session_pin, f"SESSION_START:{self.current_class_name}"))
        self.conn.commit()
        self.btn_t_break.config(state=tk.NORMAL, bg="#E63946")

    def teacher_toggle_break(self):
        if not self.is_break_active:
            self.is_break_active = True
            self.break_time_left = 600
            self.btn_t_break.config(text="Extend Break Time (+2m)", bg="#FFB703")
            self.cursor.execute("INSERT INTO chat_logs (session_id, sender, recipient, message) VALUES (?, 'SYSTEM', 'BROADCAST', 'STATE:BREAK_START')", (self.active_session_pin,))
            self.conn.commit()
            self.push_global_broadcast({"action": "BREAK_START", "duration": self.break_time_left})
            threading.Thread(target=self.teacher_break_clock_loop, daemon=True).start()
        else:
            self.break_time_left += 120
            self.push_global_broadcast({"action": "BREAK_EXTEND", "duration": self.break_time_left})

    def teacher_break_clock_loop(self):
        while self.break_time_left > 0 and self.is_break_active:
            time.sleep(1)
            self.break_time_left -= 1
        if self.break_time_left <= 0:
            self.is_break_active = False
            self.cursor.execute("INSERT INTO chat_logs (session_id, sender, recipient, message) VALUES (?, 'SYSTEM', 'BROADCAST', 'STATE:BREAK_END')", (self.active_session_pin,))
            self.conn.commit()
            self.push_global_broadcast({"action": "BREAK_END"})
            self.root.after(0, lambda: self.btn_t_break.config(text="Initiate Break Time (10m)", bg="#E63946"))

    def dispatch_student_emote(self, emote, action_name):
        lbl = tk.Label(self.s_stage, text=emote, font=("Helvetica", 36), bg="#1C2541", fg="#5BC0BE")
        lbl.place(relx=0.8, rely=0.7)
        self.root.after(2000, lbl.destroy)
        
        def push_signal():
            try:
                requests.post(SECURE_CLOUD_ENDPOINT, json={
                    "type": "STUDENT_ACTION",
                    "name": self.student_name,
                    "code": self.active_session_pin,
                    "action": action_name,
                    "timestamp": int(time.time() * 1000)
                }, timeout=3)
            except Exception: pass
        threading.Thread(target=push_signal, daemon=True).start()

    def student_sync_thread(self):
        last_seen_id = 0
        while True:
            time.sleep(1.5)
            self.cursor.execute("SELECT id, sender, recipient, message FROM chat_logs WHERE id > ? AND session_id=?", (last_seen_id, self.active_session_pin))
            entries = self.cursor.fetchall()
            
            for eid, sender, recipient, msg in entries:
                last_seen_id = eid
                if recipient == "BROADCAST":
                    if msg == "STATE:BREAK_START":
                        self.root.after(0, lambda: self.student_render_break(600))
                    elif msg == "STATE:BREAK_END":
                        self.root.after(0, self.student_clear_break)
                    elif msg.startswith("EXAM_LAUNCH:"):
                        payload = msg.split(":", 1)[1]
                        self.root.after(0, self.student_force_cbt, payload)
                elif recipient == self.student_name and sender == "TEACHER":
                    pass

    def student_render_break(self, duration_secs):
        self.s_break_overlay = tk.Frame(self.root, bg="#0B132B")
        self.s_break_overlay.place(x=0, y=0, relwidth=1, relheight=1)
        
        self.lbl_break_timer = tk.Label(self.s_break_overlay, text="Class is on a Short Break\n\nCamera System & Microphones Muted For Privacy\n\nCountdown: 10:00", font=("Helvetica", 16, "bold"), bg="#0B132B", fg="#FFB703")
        self.lbl_break_timer.pack(expand=True)
        
        self.lbl_av_indicator.config(text="🎤 Mic: MUTED BY TEACHER | 📷 Camera: BLOCKED", fg="#E63946")
        
        def count_loop(rem):
            while rem > 0 and hasattr(self, 's_break_overlay') and self.s_break_overlay.winfo_exists():
                mins, secs = divmod(rem, 60)
                try:
                    self.lbl_break_timer.config(text=f"Class is on a Short Break\n\nCamera System & Microphones Muted For Privacy\n\nCountdown: {mins:02d}:{secs:02d}")
                except Exception: break
                time.sleep(1)
                rem -= 1
        threading.Thread(target=count_loop, args=(duration_secs,), daemon=True).start()
