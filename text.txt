    def ensure_json_notes_exists(self):
        if not os.path.exists(NOTES_JSON_FILE):
            with open(NOTES_JSON_FILE, "w") as f:
                json.dump({}, f)

    def start_cloud_polling_engine(self):
        while True:
            time.sleep(2.0)
            try:
                response = requests.get(SECURE_CLOUD_ENDPOINT, timeout=3)
                if response.status_code == 200 and response.text.strip():
                    request_data = response.json()
                    req_time = request_data.get("timestamp", 0)
                    
                    if req_time > self.last_processed_time:
                        self.last_processed_time = req_time
                        req_type = request_data.get("type")
                        
                        if req_type == "TEACHER_LOGIN":
                            self.root.after(0, self.handle_web_teacher_auth, request_data)
                        elif req_type == "STUDENT_LOGIN":
                            self.root.after(0, self.handle_web_student_auth, request_data)
                        elif req_type == "STUDENT_ACTION" and request_data.get("code") == self.active_session_pin:
                            self.root.after(0, self.process_web_student_action, request_data.get("name"), request_data.get("action"))
                        elif req_type == "CBT_SUBMISSION" and request_data.get("code") == self.active_session_pin:
                            self.root.after(0, self.process_web_cbt_submission, request_data.get("name"), request_data.get("answer"))
            except Exception:
                pass

    def handle_web_teacher_auth(self, data):
        self.student_name = data.get("username")
        if hasattr(self, 'load_frame') and self.load_frame.winfo_exists(): self.load_frame.destroy()
        if hasattr(self, 'gateway_frame') and self.gateway_frame.winfo_exists(): self.gateway_frame.destroy()
        self.build_teacher_home_hub()
        self.confirm_handshake_to_cloud(self.student_name)

    def handle_web_student_auth(self, data):
        target_code = data.get("code")
        if hasattr(self, 't_roster') and len(self.t_roster.get_children()) >= self.max_students_limit:
            return

        if hasattr(self, 't_roster') and self.t_roster.winfo_exists() and target_code == self.active_session_pin:
            self.root.after(0, self.register_live_student_node, data.get("name"))
        elif not hasattr(self, 't_roster'):
            self.student_name = data.get("name")
            self.active_session_pin = target_code
            if hasattr(self, 'gateway_frame') and self.gateway_frame.winfo_exists(): self.gateway_frame.destroy()
            self.build_student_workspace()
        self.confirm_handshake_to_cloud(data.get("name"))

    def process_web_student_action(self, name, action):
        if hasattr(self, 't_roster'):
            action_labels = {
                "EMOTE_RAISE": "🖐️ Hand Raised (Awaiting Prompt)",
                "EMOTE_CLAP": "👏 Sent Performance Applaud"
            }
            mapped_text = action_labels.get(action, "Active Workspace Feed Update")
            self.register_live_student_node(name, status=mapped_text)

    def process_web_cbt_submission(self, name, answer):
        if hasattr(self, 't_roster'):
            self.register_live_student_node(name, status="CBT Score Transmitted Securely", grade=f"Locked: {answer}")

    def confirm_handshake_to_cloud(self, username):
        def run():
            try:
                confirm_url = f"https://kvdb.io/K99vVnZgYqN7Cg8n5vP9a7/nexus_confirm_{username}"
                requests.post(confirm_url, json={"status": "ACTIVE_IN_APP"}, timeout=3)
            except Exception:
                pass
        threading.Thread(target=run, daemon=True).start()

    def push_global_broadcast(self, payload):
        def task():
            try:
                payload["timestamp"] = int(time.time() * 1000)
                requests.post(f"https://kvdb.io/K99vVnZgYqN7Cg8n5vP9a7/nexus_broadcast_{self.active_session_pin}", json=payload, timeout=3)
            except Exception:
                pass
        threading.Thread(task, daemon=True).start()

    def register_live_student_node(self, name, status="Node Hook Established", grade="No Evaluation Records"):
        if hasattr(self, 't_roster'):
            for item in self.t_roster.get_children():
                if self.t_roster.item(item)['values'][0] == name:
                    self.t_roster.item(item, values=(name, status, grade))
                    return
            self.t_roster.insert("", tk.END, values=(name, status, grade))

    def init_shared_database(self):
        self.conn = sqlite3.connect("nexuslearn_system.db", check_same_thread=False)
        self.cursor = self.conn.cursor()
        self.cursor.execute("""
            CREATE TABLE IF NOT EXISTS teachers (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                username TEXT UNIQUE NOT NULL,
                password TEXT NOT NULL,
                auth_pin TEXT UNIQUE NOT NULL,
                payment_status TEXT DEFAULT 'Unpaid',
                expiration_timestamp DATETIME
            )
        """)
        self.cursor.execute("""
            CREATE TABLE IF NOT EXISTS chat_logs (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                session_id TEXT, sender TEXT, recipient TEXT, message TEXT, timestamp DATETIME DEFAULT CURRENT_TIMESTAMP
            )
        """)
        self.conn.commit()

            self.wave_canvas.delete("all")

    def save_notes_json(self):
        title = self.ent_note_title.get().strip() or "Untitled"
        body = self.hub_notepad.get("1.0", tk.END).strip()
        try:
            with open(NOTES_JSON_FILE, "r") as f: data = json.load(f)
            data[title] = body
            with open(NOTES_JSON_FILE, "w") as f: json.dump(data, f, indent=4)
        except Exception: pass

    def load_notes_json(self):
        title = self.ent_note_title.get().strip()
        try:
            with open(NOTES_JSON_FILE, "r") as f: data = json.load(f)
            if title in data:
                self.hub_notepad.delete("1.0", tk.END)
                self.hub_notepad.insert(tk.END, data[title])
        except Exception: pass
